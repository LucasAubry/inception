  -newkey rsa:4096 \
  -keyout srcs/requirements/nginx/conf/certs/yourlogin.42.fr.key \
  -out srcs/requirements/nginx/conf/certs/yourlogin.42.fr.crt \
  -subj "/CN=yourlogin.42.fr/O=42/"
